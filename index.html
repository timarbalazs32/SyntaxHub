Backend Documentation

Node.js + SQLite3 database solutions for modern APIs.

1. SQLite3 Example (Async / Callback)

Using sqlite3 driver in Express.js with ESM syntax and callback pattern.

index.js (sqlite3) 
import express from 'express';
import cors from 'cors';
import sqlite3 from 'sqlite3';

const app = express();
const port = 8000;
app.use(cors());
app.use(express.json());

const db = new sqlite3.Database(`sakila.db`);

app.get('/', (req, res) => {
    res.json({message: 'Welcome to Film Portal API'});
});

app.get('/films', (req, res) => {
    let {limit = 50, offset = 0} = req.query;
    const sql = `SELECT * FROM film LIMIT ? OFFSET ?`;
    db.all(sql, [limit, offset], (err, rows) => {
        if (err) return res.status(400).json({error: err.message});
        res.json(rows);
    });
});

app.get('/films/:id', (req, res) => {
    const {id} = req.params;
    db.get(`SELECT * FROM film WHERE film_id = ?`, [id], (err, row) => {
        if (err) return res.status(404).json({error: err.message});
        row ? res.json(row) : res.status(404).json({error: "Film not found"});
    });
});

app.post('/films', (req, res) => {
    const {film_id, title, description, release_year, language_id, rental_duration, rental_rate, length, replacement_cost, rating, special_features} = req.body;
    const sql = `INSERT INTO film VALUES (?,?,?,?,?,?,?,?,?,?,?, datetime('now'))`;
    db.run(sql, [film_id, title, description, release_year, language_id, rental_duration, rental_rate, length, replacement_cost, rating, special_features], function(err) {
        if (err) return res.status(400).json({error: err.message});
        res.status(201).json({id: this.lastID});
    });
});

app.listen(port, () => console.log(`Server on port ${port}`));

2. Better-SQLite3 (Synchronous / Cinema API)

Using better-sqlite3 for synchronous operations. Methods: .all() for many, .get() for one, .run() for mutations.

index.js (better-sqlite3) 
const cors = require('cors');
const express = require('express');
const Database = require('better-sqlite3');

const app = express();
app.use(express.json());
app.use(cors());

const db = new Database('mozihalo.db');

app.get("/filmek", (req, res) => {
  try {
    const films = db.prepare("SELECT * FROM filmek").all();
    res.json(films);
  } catch (error) { res.status(500).json(error.message); }
});

app.get("/vetites/:vazon", (req, res) => {
  const result = db.prepare(`SELECT filmek.fcim, mozik.mnev, mozik.varos 
    FROM vetitesek JOIN mozik ON mozik.mazon = vetitesek.mazon 
    JOIN filmek ON filmek.fazon = vetitesek.fazon WHERE vetitesek.vazon = ?`).get(req.params.vazon);
  res.json(result);
});

app.post("/filmek", (req, res) => {
  const {fazon, fcim, rendezo, gyartasi_ev, mufaj, jegypont} = req.body;
  const info = db.prepare("INSERT INTO filmek VALUES (?,?,?,?,?,?)").run(fazon, fcim, rendezo, gyartasi_ev, mufaj, jegypont);
  res.status(201).json({id: info.lastInsertRowid});
});

app.delete("/film/:fazon", (req, res) => {
  db.prepare("DELETE FROM filmek WHERE fazon = ?").run(req.params.fazon);
  res.json({message: "Deleted"});
});

app.listen(8000, () => console.log("Server running"));

2. Student Service & Components

tanulo.service.ts
import { Injectable } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { Observable } from 'rxjs';

export interface Tanulo {
  oktazon: number;
  nev: string;
  hozott: number;
  kpmagy: number;
  kpmat: number;
}

@Injectable({providedIn: 'root'})
export class TanuloService {
  private apiUrl = 'http://localhost:8000/tanulok';
  constructor(private http: HttpClient) {}

  getOsszesTanulo(): Observable {
    return this.http.get(this.apiUrl);
  }

  deleteTanulo(oktazon: number): Observable {
    return this.http.delete(`http://localhost:8000/tanulo/id/${oktazon}`);
  }

  addTanulo(tanulo: Tanulo): Observable {
    return this.http.post("http://localhost:8000/tanulo", tanulo);
  }
}
